  return r.value;
}

document.querySelectorAll('.ui-code.line-numbers').forEach(function(block) {
  var code = block.querySelector('pre code');
  var col = block.querySelector('.line-numbers-col');
  if (!code || !col) return;

});

// Toggle popover on button click
document.querySelectorAll('.ui-popover').forEach(btn => {
  btn.addEventListener('click', () => {
    btn.classList.toggle('open');
  });
});

// Dismiss when clicking outside
document.addEventListener('click', e => {
  document.querySelectorAll('.ui-popover.open').forEach(p => {
    if (!p.contains(e.target)) p.classList.remove('open');
  });
});
</script>
