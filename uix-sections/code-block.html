With filename & line numbers

src/utils.ts 
export type Result<T> =
  | { ok: true; value: T }
  | { ok: false; error: Error };

export function unwrap<T>(r: Result<T>): T {
  if (!r.ok) throw r.error;
  return r.value;
}
